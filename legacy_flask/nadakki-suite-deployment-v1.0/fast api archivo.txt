def archive_fastapi():
    """Archive FastAPI code for future v2.0 reference"""
    fastapi_dir = ROOT / "backend" / "fastapi"
    archive_dir = ROOT / "archive" / "fastapi_reference_2025"

        import shutil
        shutil.move(str(fastapi_dir), str(archive_dir))
        print(f"‚úÖ FastAPI code archived to: {archive_dir}")
    else:
        print("‚ö†Ô∏è  No FastAPI directory found")
